        // Load question function
        function loadQuestion(index) {
            const question = examData[index];
            
            // Update UI elements
            questionNumber.textContent = index + 1;
            currentQuestionEl.textContent = index + 1;
            questionType.textContent = question.type === 'mcq' ? 'Multiple Choice' : 'True/False';
            questionText.textContent = question.question;
            topicIndicator.textContent = question.topic;
            
            // Show/hide appropriate answer containers
            if (question.type === 'mcq') {
                optionsContainer.style.display = 'grid';
                tfContainer.style.display = 'none';
                
                // Clear and rebuild options
                optionsContainer.innerHTML = '';
                question.options.forEach((optionText, i) => {
                    const option = document.createElement('div');
                    option.className = 'option';
                    option.textContent = optionText;
                    option.dataset.value = String.fromCharCode(65 + i); // A, B, C, D
                    
                    // Restore previous answer if exists
                    if (userAnswers[index] === option.dataset.value) {
                        option.classList.add('selected');
                    }
                    
                    option.addEventListener('click', () => handleOptionSelect(option, index));
                    optionsContainer.appendChild(option);
                });
            } else {
                optionsContainer.style.display = 'none';
                tfContainer.style.display = 'flex';
                
                // Restore previous answer for TF
                const tfOptions = tfContainer.querySelectorAll('.tf-option');
                tfOptions.forEach(option => {
                    option.classList.remove('selected', 'correct', 'incorrect');
                    if (userAnswers[index] === option.dataset.value) {
                        option.classList.add('selected');
                    }
                    option.onclick = () => handleTFSelect(option, index);
                });
            }
            
            // Show/hide feedback based on whether question was answered
            if (userAnswers[index] !== null) {
                showFeedback(index);
            } else {
                feedbackArea.style.display = 'none';
                // Reset feedback container state
                delete feedbackArea.dataset.answered;
            }
            
            // Update navigation buttons
            prevButton.disabled = index === 0;
            nextButton.textContent = index === totalQuestions - 1 ? 'Finish Examination' : 'Next Question →';
            nextButton.className = index === totalQuestions - 1 ? 'nav-button finish' : 'nav-button next';
            
            // Scroll to top of question
            questionCard.scrollIntoView({ behavior: 'smooth', block: 'start' });
            
            // Update progress bar
            updateProgress();
        }

        // Handle MCQ option selection
        function handleOptionSelect(option, questionIndex) {
            const question = examData[questionIndex];
            
            // Remove selection states from all options
            const allOptions = optionsContainer.querySelectorAll('.option');
            allOptions.forEach(opt => {
                opt.classList.remove('selected', 'correct', 'incorrect');
            });
            
            // Add selected state to clicked option
            option.classList.add('selected');
            
            // Store answer
            userAnswers[questionIndex] = option.dataset.value;
            
            // Show feedback
            showFeedback(questionIndex);
        }

        // Handle True/False selection
        function handleTFSelect(option, questionIndex) {
            const question = examData[questionIndex];
            
            // Remove selection states from all options
            const allOptions = tfContainer.querySelectorAll('.tf-option');
            allOptions.forEach(opt => {
                opt.classList.remove('selected', 'correct', 'incorrect');
            });
            
            // Add selected state to clicked option
            option.classList.add('selected');
            
            // Store answer
            userAnswers[questionIndex] = option.dataset.value;
            
            // Show feedback
            showFeedback(questionIndex);
        }

        // Show feedback after answering
        function showFeedback(questionIndex) {
            const question = examData[questionIndex];
            const userAnswer = userAnswers[questionIndex];
            const isCorrect = userAnswer === question.correct;
            
            // Only update score once per question
            if (!feedbackArea.dataset.answered && userAnswer !== null) {
                if (isCorrect) {
                    score++;
                    currentScoreEl.textContent = score;
                }
                feedbackArea.dataset.answered = 'true';
            }
            
            // Show feedback UI
            feedbackArea.style.display = 'block';
            feedbackText.textContent = isCorrect ? 
                '✓ Absolutely Correct!' : 
                `✗ Incorrect Answer • Correct answer: ${question.correct === 'true' ? 'TRUE' : question.correct === 'false' ? 'FALSE' : question.correct}`;
            
            feedbackText.className = isCorrect ? 
                'feedback-text feedback-correct' : 
                'feedback-text feedback-incorrect';
            
            // Highlight correct answer if user was wrong
            if (!isCorrect) {
                if (question.type === 'mcq') {
                    const correctOption = Array.from(optionsContainer.children).find(
                        opt => opt.dataset.value === question.correct
                    );
                    if (correctOption) correctOption.classList.add('correct');
                } else {
                    const correctOption = tfContainer.querySelector(`.tf-option[data-value="${question.correct}"]`);
                    if (correctOption) correctOption.classList.add('correct');
                }
            }
            
            // Update progress immediately after answering
            updateProgress();
        }

        // Update progress bar
        function updateProgress() {
            const answeredCount = userAnswers.filter(answer => answer !== null).length;
            const progressPercent = Math.min(100, Math.round((answeredCount / totalQuestions) * 100));
            progressBar.style.width = `${progressPercent}%`;
        }
